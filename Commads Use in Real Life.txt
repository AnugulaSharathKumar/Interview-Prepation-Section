🧩 1. LINUX — Troubleshooting, Networking & Performance
🔍 System Information & Monitoring
uname -a                      # Kernel & system info
cat /etc/*release             # OS version
uptime                        # System load averages
top / htop                    # Real-time process view
free -h                       # Memory usage
vmstat 1                      # CPU, I/O, memory usage live
iostat -xz 1                  # I/O stats
mpstat -P ALL 1               # CPU utilization per core
lscpu                         # CPU details
lsblk                         # Block devices
df -hT                        # Disk space with filesystem type
du -sh *                      # Disk usage per folder

🌐 Network Diagnostics
ip a                          # Show IP addresses
ip r                          # Show routes
ss -tulwn                     # Listening ports
netstat -plant                # Active connections
ping -c4 8.8.8.8              # Ping test
curl -I http://example.com    # Check HTTP headers
wget --spider example.com     # Check site reachability
dig example.com +short        # DNS lookup
traceroute google.com         # Route trace
telnet <host> <port>          # Port check
nc -zv <ip> <port>            # Network connectivity test
arp -n                        # Show ARP table

🪵 Logs & Files
tail -f /var/log/syslog                   # Live system log
grep -i error /var/log/syslog             # Search logs
journalctl -u nginx.service               # Systemd service logs
journalctl -xe                            # View recent errors
find / -type f -name "*.log"              # Find log files
awk '{print $1,$2,$3}' /var/log/syslog   # Extract fields
sed -i 's/old/new/g' file.txt             # Replace text inline

⚙️ System & Service Management
systemctl status docker                   # Check service status
systemctl restart nginx                   # Restart service
systemctl enable nginx                    # Enable on boot
ps aux | grep java                        # Process search
kill -9 <pid>                             # Kill process
crontab -l                                # List cron jobs
dmesg | tail                              # Kernel logs

🧱 2. GIT — Collaboration & Code Control
🔄 Basic Workflow
git clone <repo>                        # Clone repo
git status                              # View changes
git add .                               # Stage all files
git commit -m "message"                 # Commit
git push origin main                    # Push
git pull origin main                    # Pull

🌿 Branching & Merging
git branch                              # List branches
git checkout -b feature/new             # New branch
git merge feature/new                   # Merge branch
git rebase main                         # Rebase
git stash / git stash pop               # Temporarily save work

🧰 Troubleshooting
git log --oneline --graph --decorate    # Compact commit log
git diff                                # Compare changes
git reset --hard HEAD~1                 # Undo last commit
git reflog                              # Recover deleted commits
git remote -v                           # Show remotes
git config --global user.name "Sharath"
git config --global user.email "email"

🐳 3. DOCKER — Containers & Images
📦 Containers
docker ps -a                            # List containers
docker logs -f <container>              # Follow logs
docker exec -it <container> bash        # Access container
docker inspect <container>              # Inspect details
docker stats                            # Show usage
docker top <container>                  # Show processes
docker restart <container>              # Restart container

🧰 Images & Cleanup
docker images                           # List images
docker rmi <image_id>                   # Delete image
docker build -t myapp:latest .          # Build image
docker pull <image>                     # Pull image
docker system prune -a                  # Clean unused data
docker volume prune                     # Clean volumes

🛠️ Networking & Compose
docker network ls                       # List networks
docker network inspect bridge            # Inspect network
docker-compose up -d                    # Run compose
docker-compose logs -f                  # View compose logs
docker-compose down                     # Stop and remove

☸️ 5. KUBERNETES — Cluster Operations & Debugging
📦 Pods & Containers
kubectl get pods -A                     # List all pods
kubectl describe pod <pod>              # Pod details
kubectl logs -f <pod>                   # Live logs
kubectl exec -it <pod> -- bash          # Shell inside pod
kubectl delete pod <pod>                # Delete pod
kubectl top pod                         # Pod resource usage

🚀 Deployments & Services
kubectl get deployments                 # List deployments
kubectl rollout status deploy/<name>    # Rollout progress
kubectl rollout undo deploy/<name>      # Rollback
kubectl scale deploy/<name> --replicas=3
kubectl get svc                         # Services
kubectl port-forward svc/myapp 8080:80  # Local access

⚙️ Config & Nodes
kubectl get nodes -o wide               # Node details
kubectl describe node <node>            # Node info
kubectl get ns                          # Namespaces
kubectl get configmap -A                # ConfigMaps
kubectl get secret -A                   # Secrets
kubectl get ingress -A                  # Ingresses
kubectl get pvc -A                      # Persistent volumes

🧰 Troubleshooting
kubectl get events --sort-by=.metadata.creationTimestamp
kubectl logs -f -n <ns> --all-containers=true
kubectl cp <pod>:/tmp/log.txt ./        # Copy from pod
kubectl debug node/<node> --image=busybox -it
kubectl describe service <svc>
